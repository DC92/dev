src/layerSwitcher.js add a tinny efficient layer switcher for both baselayers & overlays.

new ol.Map({
	target: 'map',
	view: new ol.View({
		center: [700000, 5600000],
		zoom: 7,
	}),
	controls: [
		new ol.control.Zoom(),
		controlLayerSwitcher({
			baseLayers: {
				OSM: new ol.layer.Tile({
					source: new ol.source.OSM(),
				}),
				Stamen: new ol.layer.Tile({
					source: new ol.source.Stamen({
						layer: 'terrain',
					}),
				}),
				Watercolor: new ol.layer.Tile({
					source: new ol.source.Stamen({
						layer: 'watercolor',
					}),
				}),
			},
		}),
	],
});
